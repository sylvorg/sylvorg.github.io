setFail () { set -eo pipefail; }
ensure () {
    hash "$1" 2> /dev/null
    if [ $? -ne 0 ]; then
	hash nix 2> /dev/null
	[ $? -ne 0 ] && sh <(curl -fsSL https://nixos.org/nix/install) --daemon
	[ -z "$2" ] && nix-env -i "$1" || nix-env -i "$2"
    fi
}
remove () { [ -z "$2" ] && nix-env -e "$1" || nix-env -e "$2"; }

for p in yadm git "make gnumake" fd emacs; do ensure $p; done

read -p $'Impermanent?\n' impermanent < /dev/tty
if [ "$impermanent" == "y" ] || [ "$impermanent" == "yes" ]; then
    worktree="/persist/$HOME"
else
    worktree="$HOME"
fi

read -p $'Yadm clone?\n' yc < /dev/tty
cloneCommand="-w $worktree --no-bootstrap https://github.com/shadowrylander/shadowrylander.git"
if [ "$yc" == "y" ] || [ "$yc" == "yes" ]; then
    if [ -d "$HOME/.local/share/yadm/repo.git" ]; then
	read -p $'Repo already exists! Force clone?\n' fc < /dev/tty
	if [ "$fc" == "y" ] || [ "$fc" == "yes" ]; then
	    yadm clone -f $cloneCommand
	fi
    else
	yadm clone $cloneCommand
    fi

    yadm remote set-url --push origin git@github.com:shadowrylander/shadowrylander.git
    yadm gitconfig core.worktree $HOME

    # TODO: May need `yadm stash' here
    # yadm stash
fi

nixos-generate-config || :

read -p $'Tell me your secrets!\n' secrets < /dev/tty
secCloneCommand="-w $worktree --no-bootstrap $secrets"
if [ "$secrets" != "n" ] || [ "$secrets" != "no" ]; then
    if [ -d "$HOME/.local/share/yadm-sec/repo.git" ]; then
	read -p $'Secrets repo already exists! Force clone secrets?\n' forceSecrets < /dev/tty
	if [ "$forceSecrets" == "y" ] || [ "$forceSecrets" == "yes" ]; then
	    yadm sec clone -f $secCloneCommand
	fi
    else
	yadm sec clone $secCloneCommand
    fi

    yadm sec gitconfig core.worktree $HOME

    # TODO: May need `yadm sec stash' here
    # yadm sec stash

    chmod +x "$HOME/.config/yadm-sec/bootstrap"
    "$HOME/.config/yadm-sec/bootstrap"
fi

read -p $'Initialize Yadm Submodules?\n' ys < /dev/tty
if [ "$ys" == "y" ] || [ "$ys" == "yes" ]; then
    GIT_SSL_NO_VERIFY=1 GIT_DISCOVERY_ACROSS_FILESYSTEM=1 yadm submodule update --init --depth 1 --recursive --remote --force
    make -f "$HOME/.emacs.d/makefile" soft-init
fi

read -p $'Bootstrap?\n' bootstrap < /dev/tty
if [ "$bootstrap" == "y" ] | [ "$bootstrap" == "yes" ]; then
    chmod +x "$HOME/.config/yadm/bootstrap"
    "$HOME/.config/yadm/bootstrap"
fi

.RECIPEPREFIX := |
.DEFAULT_GOAL := super-push

# Adapted From: https://www.systutorials.com/how-to-get-the-full-path-and-directory-of-a-makefile-itself/
mkfilePath := $(abspath $(lastword $(MAKEFILE_LIST)))
mkfileDir := $(dir $(mkfilePath))

et:
|chmod +x $(mkfileDir)/settings/*.sh
|$(mkfileDir)/settings/org-export.sh $(mkfileDir)/README.org
|$(mkfileDir)/settings/org-tangle.sh $(mkfileDir)/README.org

commit:
|git -C $(mkfileDir) commit --allow-empty-message -am ""

push:
|git -C $(mkfileDir) push

cpush: commit push

super-push: et cpush
