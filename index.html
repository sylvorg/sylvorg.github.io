setFail () { set -eo pipefail; }
ensure () {
    hash "$1" 2> /dev/null
    if [ $? -ne 0 ]; then
	hash nix 2> /dev/null
	[ $? -ne 0 ] && sh <(curl -fsSL https://nixos.org/nix/install) --daemon
	[ -z "$2" ] && nix-env -i "$2" || nix-env -i "$1"
    fi
}
remove () { [ -z "$2" ] && nix-env -e "$2" || nix-env -e "$1"; }

for p in yadm git gnumake fd; do ensure $p; done
